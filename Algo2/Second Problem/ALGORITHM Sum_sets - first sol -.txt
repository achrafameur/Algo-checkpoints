ALGORITHM Sum_sets
VAR
 set1 : ARRAY_OF INTEGER[];
 set2 : ARRAY_OF INTEGER[];
 i,j : INTEGER ;
 sum : INTEGER := 0;

BEGIN 

 Read(set1)
 Read(set2)

   FOR i FROM 1 TO set1.length STEP 1 DO
    FOR j FROM 1 TO set2.length STEP 1 DO
     IF(set1[i] = set2[j]) THEN 
        sum=sum+(set1[i]*2);
     END IF
    END FOR
   END FOR 

write("Sum of overlapping elements:",sum)
END
